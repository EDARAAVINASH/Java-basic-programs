#include<stdio.h>
int main()
{
    int i,j,n;
    scanf("%d",&n);
    for(i=0;i<(n/2)+1;i++)
    {
        for(j=i;j<n/2;j++)
        {
            printf("_");
        }
        for(j=0;j<=2*i;j++)
        {
            printf("*");
        }
        printf("\n");
    }
    for(i=0;i<n/2;i++)
    {
        for(j=0;j<=i;j++)
        {
            printf("_");
        }
        for(j=0;j<n/2+i-;j++)
        {
            printf("*");
        }
        printf("\n");
    }
}
